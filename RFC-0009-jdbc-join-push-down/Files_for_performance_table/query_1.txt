Select
  A.ASSIGNMENT_ID,
  A.ROLE,
  A.IS_CURRENT,
  B.CUST_ID,
  B.FIRST_NAME,
  B.LAST_NAME,
  C.ASSIGNMENT_ID,
  C.DEFAULT_INT
FROM
  DB2.DB2.ASSIGNMENTS_10_MILLION A
  JOIN DB2.DB2.CUSTOMER_10_MILLION B ON A.EMPLOYEE_ID = B.CUST_ID
  JOIN DB2.DB2.JOIN_TABLE_50_ROWS C ON A.ASSIGNMENT_ID = C.ASSIGNMENT_ID;
